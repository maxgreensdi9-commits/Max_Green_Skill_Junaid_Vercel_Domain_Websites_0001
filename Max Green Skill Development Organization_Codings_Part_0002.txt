    // Process voice/text commands
    function processVoiceCommand(command) {
      const lowerCommand = command.toLowerCase();
      
      // Show typing indicator
      const typingIndicator = document.createElement('div');
      typingIndicator.className = 'question-answer';
      typingIndicator.innerHTML = `<div class="answer"><span class="typing-indicator"><span></span><span></span><span></span></span> Thinking...</div>`;
      voiceOutput.appendChild(typingIndicator);
      voiceOutput.scrollTop = voiceOutput.scrollHeight;
      
      // Simulate AI response with a slight delay
      setTimeout(() => {
        // Remove typing indicator
        voiceOutput.removeChild(typingIndicator);
        
        let response = "I'm not sure how to help with that. Try asking about our monitoring visits, programs, or contact information.";
        
        // Check for exact matches first
        if (qaDatabase[command]) {
          response = qaDatabase[command].answer;
        } else {
          // Check for partial matches
          for (const [key, value] of Object.entries(qaDatabase)) {
            if (lowerCommand.includes(key.toLowerCase())) {
              response = value.answer;
              break;
            }
          }
          
          // Fallback responses for general queries
          if (lowerCommand.includes('hello') || lowerCommand.includes('hi')) {
            response = "Hello! I'm the MGSDO assistant. How can I help you learn about our organization?";
          } else if (lowerCommand.includes('monitoring') || lowerCommand.includes('visit')) {
            response = "Our team recently conducted surprise monitoring visits to NFE Centres in Tharparkar. The visits aimed to ensure quality education and community participation in our initiatives.";
          } else if (lowerCommand.includes('program') || lowerCommand.includes('course') || lowerCommand.includes('nfe')) {
            response = "We operate Non-Formal Education Centres across Tharparkar with support from the Government of Sindh. Our programs focus on basic education and skill development for children and adults.";
          } else if (lowerCommand.includes('contact') || lowerCommand.includes('email') || lowerCommand.includes('phone')) {
            response = "You can contact us at: info@maxgreenskills.org or call +92 333 3900728. Our head office is in Ghazi Goth, Karachi.";
          } else if (lowerCommand.includes('location') || lowerCommand.includes('address') || lowerCommand.includes('tharparkar')) {
            response = "Our NFE Centres are located across District Tharparkar. Our head office is at Shaheed Makhdoom Bilawal Society, Ghazi Goth, Karachi.";
          } else if (lowerCommand.includes('mission') || lowerCommand.includes('vision') || lowerCommand.includes('purpose')) {
            response = "Our mission is to empower underprivileged communities through education and skill development, with a focus on non-formal education in remote areas like Tharparkar.";
          } else if (lowerCommand.includes('help')) {
            response = "I can help you learn about MGSDO's monitoring visits, programs, contact information, location, and mission. Just ask!";
          } else if (lowerCommand.includes('link') || lowerCommand.includes('website')) {
            response = "We have multiple websites and deployment links. You can visit maxgreenskills.de, maxgreenskills.org, or any of our Netlify deployment sites. Check the links section for all available URLs.";
          } else if (lowerCommand.includes('facebook')) {
            response = "You can find us on Facebook at facebook.com/profile.php?id=100065260100181. We have photos, reels, and information about our activities there.";
          } else if (lowerCommand.includes('leadership') || lowerCommand.includes('team')) {
            response = "Our leadership team includes CEO Munsif Ali Lashari, Director Imtiaz Ali Chandio, Director Admin Hamza Khalid Arain, Director Accounts Ana Sehar, and Director Media Tariq Anwar.";
          }
        }
        
        addToOutput(`Assistant: ${response}`);
        
        // Speak the response if speech synthesis is available
        if ('speechSynthesis' in window) {
          const utterance = new SpeechSynthesisUtterance(response);
          utterance.rate = 0.9;
          utterance.pitch = 1;
          window.speechSynthesis.speak(utterance);
        }
      }, 1500);
    }

    // Initialize with welcome message
    window.addEventListener('load', function() {
      if ('speechSynthesis' in window) {
        const welcomeMsg = new SpeechSynthesisUtterance("Welcome to Max Green Skills Development Organization. How can I assist you today?");
        welcomeMsg.rate = 0.9;
        window.speechSynthesis.speak(welcomeMsg);
      }
    });
  </script>
