  <script>
    // Voice Assistant Elements
    const voiceBtn = document.getElementById('voiceBtn');
    const voicePanel = document.getElementById('voicePanel');
    const closePanel = document.getElementById('closePanel');
    const voiceOutput = document.getElementById('voiceOutput');
    const textInput = document.getElementById('textInput');
    const textSubmit = document.getElementById('textSubmit');

    // Speech Recognition Setup
    const SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition;
    let recognition;
    let isListening = false;

    if (SpeechRecognition) {
      recognition = new SpeechRecognition();
      recognition.continuous = false;
      recognition.interimResults = false;
      recognition.lang = 'en-US';

      recognition.onstart = function() {
        isListening = true;
        voiceBtn.classList.add('listening');
        addToOutput("Listening...");
      };

      recognition.onresult = function(event) {
        const transcript = event.results[0][0].transcript;
        addToOutput(`You said: ${transcript}`);
        processVoiceCommand(transcript);
      };

      recognition.onerror = function(event) {
        addToOutput(`Error: ${event.error}`);
        isListening = false;
        voiceBtn.classList.remove('listening');
      };

      recognition.onend = function() {
        isListening = false;
        voiceBtn.classList.remove('listening');
      };
    } else {
      addToOutput("Speech recognition not supported in this browser.");
    }

    // Toggle voice recognition
    voiceBtn.addEventListener('click', function() {
      if (!SpeechRecognition) {
        addToOutput("Speech recognition is not supported in your browser.");
        return;
      }

      if (isListening) {
        recognition.stop();
      } else {
        recognition.start();
      }
      
      // Show the voice panel when starting to listen
      if (!voicePanel.classList.contains('active')) {
        voicePanel.classList.add('active');
      }
    });

    // Close voice panel
    closePanel.addEventListener('click', function() {
      voicePanel.classList.remove('active');
      if (isListening) {
        recognition.stop();
      }
    });

    // Text input submission
    textSubmit.addEventListener('click', function() {
      const text = textInput.value.trim();
      if (text) {
        addToOutput(`You typed: ${text}`);
        processVoiceCommand(text);
        textInput.value = '';
      }
    });

    textInput.addEventListener('keypress', function(e) {
      if (e.key === 'Enter') {
        textSubmit.click();
      }
    });

    // Add message to output
    function addToOutput(message) {
      const p = document.createElement('p');
      p.textContent = message;
      voiceOutput.appendChild(p);
      voiceOutput.scrollTop = voiceOutput.scrollHeight;
    }

    // Process voice/text commands
    function processVoiceCommand(command) {
      const lowerCommand = command.toLowerCase();
      
      // Simulate AI response with a slight delay
      setTimeout(() => {
        let response = "I'm not sure how to help with that. Try asking about our monitoring visits, programs, or contact information.";
        
        if (lowerCommand.includes('hello') || lowerCommand.includes('hi')) {
          response = "Hello! I'm the MGSDO assistant. How can I help you learn about our organization?";
        } else if (lowerCommand.includes('monitoring') || lowerCommand.includes('visit')) {
          response = "Our team recently conducted surprise monitoring visits to NFE Centres in Tharparkar. The visits aimed to ensure quality education and community participation in our initiatives.";
        } else if (lowerCommand.includes('program') || lowerCommand.includes('course') || lowerCommand.includes('nfe')) {
          response = "We operate Non-Formal Education Centres across Tharparkar with support from the Government of Sindh. Our programs focus on basic education and skill development for children and adults.";
