        let speechSynthesis = window.speechSynthesis;
        let currentUtterance = null;
        let isSpeaking = false;
        let recognition;

        // Initialize speech recognition if available
        if ('webkitSpeechRecognition' in window || 'SpeechRecognition' in window) {
            const SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition;
            recognition = new SpeechRecognition();
            recognition.continuous = false;
            recognition.interimResults = false;
            recognition.lang = 'en-US';

            recognition.onresult = function(event) {
                const transcript = event.results[0][0].transcript.toLowerCase();
                document.getElementById('questionInput').value = transcript;
                askQuestion(transcript);
            };

            recognition.onerror = function(event) {
                console.error('Speech recognition error', event.error);
                document.getElementById('status').textContent = 'Error: ' + event.error;
            };

            recognition.onend = function() {
                document.getElementById('status').textContent = 'Ready to answer questions';
            };
        }

        function startListening() {
            if (recognition) {
                document.getElementById('status').textContent = 'Listening... Speak now';
                recognition.start();
            } else {
                document.getElementById('status').textContent = 'Speech recognition not supported in this browser';
            }
        }

        function askQuestion(question) {
            if (!question.trim()) return;
            
            // Add user question to chat
            addMessageToChat(question, 'user');
            
            // Find the best matching answer
            const normalizedQuestion = question.toLowerCase().trim();
            let answer = faqDatabase.default.answer;
            
            for (const [key, value] of Object.entries(faqDatabase)) {
                if (normalizedQuestion.includes(key)) {
                    answer = value.answer;
                    break;
                }
            }
            
            // Special handling for CEO name variations
            if (normalizedQuestion.includes('munsif') || normalizedQuestion.includes('lashari') || 
                normalizedQuestion.includes('ceo') || normalizedQuestion.includes('leader')) {
                answer = faqDatabase["who is the ceo"].answer;
            }
            
            // Add assistant answer to chat
            addMessageToChat(answer, 'assistant');
            
            // Speak the answer
            speakText(answer);
        }

        function addMessageToChat(message, sender) {
            const chatContainer = document.getElementById('chatContainer');
            const messageElement = document.createElement('div');
            messageElement.classList.add('chat-message');
            messageElement.classList.add(sender === 'user' ? 'user-message' : 'assistant-message');
            messageElement.textContent = message;
            chatContainer.appendChild(messageElement);
            chatContainer.scrollTop = chatContainer.scrollHeight;
        }

        function speakText(text) {
            if (speechSynthesis.speaking) {
                speechSynthesis.cancel();
            }

            currentUtterance = new SpeechSynthesisUtterance(text);
            
            // Configure voice settings
            const { rate, pitch } = getVoiceSettings();
            currentUtterance.rate = rate;
            currentUtterance.pitch = pitch;
            currentUtterance.volume = 1;

            // Select voice based on gender preference
            const voiceType = document.getElementById('voiceSelect').value;
            const voices = speechSynthesis.getVoices();
            const preferredVoices = voices.filter(voice => {
                return voiceType === 'female' ? 
                    voice.name.toLowerCase().includes('female') || voice.name.toLowerCase().includes('samantha') || voice.name.toLowerCase().includes('karen') :
                    voice.name.toLowerCase().includes('male') || voice.name.toLowerCase().includes('alex') || voice.name.toLowerCase().includes('daniel');
            });

            if (preferredVoices.length > 0) {
                currentUtterance.voice = preferredVoices[0];
            }

            // Event handlers
            currentUtterance.onstart = function() {
                isSpeaking = true;
                document.getElementById('status').textContent = 'Speaking...';
                document.getElementById('status').className = 'status speaking';
            };

            currentUtterance.onend = function() {
                isSpeaking = false;
                document.getElementById('status').textContent = 'Ready to answer questions';
                document.getElementById('status').className = 'status';
            };

            currentUtterance.onerror = function(event) {
                isSpeaking = false;
                document.getElementById('status').textContent = 'Error: ' + event.error;
                document.getElementById('status').className = 'status';
            };

            speechSynthesis.speak(currentUtterance);
        }

        function getVoiceSettings() {
            const rate = parseFloat(document.getElementById('rateSlider').value);
            const pitch = parseFloat(document.getElementById('pitchSlider').value);
            
            return { rate, pitch };
        }

        function stopReading() {
            if (speechSynthesis.speaking) {
                speechSynthesis.cancel();
                isSpeaking = false;
                document.getElementById('status').textContent = 'Stopped';
                document.getElementById('status').className = 'status';
            }
        }

        function updateRate() {
            const rateValue = document.getElementById('rateSlider').value;
            document.getElementById('rateValue').textContent = rateValue;
            if (isSpeaking && currentUtterance) {
                currentUtterance.rate = parseFloat(rateValue);
            }
        }

        function updatePitch() {
            const pitchValue = document.getElementById('pitchSlider').value;
            document.getElementById('pitchValue').textContent = pitchValue;
            if (isSpeaking && currentUtterance) {
                currentUtterance.pitch = parseFloat(pitchValue);
            }
        }

        // Load available voices when they are ready
        speechSynthesis.onvoiceschanged = function() {
            const voices = speechSynthesis.getVoices();
            console.log('Available voices:', voices);
        };

        // Initialize slider values
        document.getElementById('rateValue').textContent = document.getElementById('rateSlider').value;
        document.getElementById('pitchValue').textContent = document.getElementById('pitchSlider').value;

        // Allow pressing Enter in the input field
        document.getElementById('questionInput').addEventListener('keypress', function(e) {
            if (e.key === 'Enter') {
                askQuestion(this.value);
            }
        });
