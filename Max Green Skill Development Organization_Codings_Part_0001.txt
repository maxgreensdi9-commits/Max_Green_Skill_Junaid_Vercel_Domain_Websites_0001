            <h2>Surprise Monitoring Visits ‚Äì Lighting the Path of Learning in Tharparkar üåü</h2>
            <div class="content">
                <p>In a continued effort to ensure quality education and impactful learning experiences, CEO Max Green Skill Development Organisation, Mr. Munsif Lashari, and District Education Officer Literacy (DEO) Mr Kastoor Chand along with Program Manager NFE, Mr. Hamza Khalid, conducted surprise monitoring visits to various Non-Formal Education (NFE) Centres across District Tharparkar.</p>
                <p>The purpose of these visits was to review the progress of the centers, evaluate learner attendance, and observe community participation ‚Äî key elements that ensure sustainability and strengthen the spirit of local ownership in education initiatives.</p>
                <p>During the visits, the team expressed satisfaction over the dedication of learners and facilitators while encouraging continuous improvement to make learning more engaging and effective for children and adults alike.</p>
                <p>These NFE Centres are being operated under the management of Max Green Skill Development Organisation, with the valuable support of the Directorate of Literacy and Non-Formal Education, Government of Sindh.</p>

        // Text-to-Speech Functionality
        const femaleVoiceBtn = document.getElementById('femaleVoice');
        const maleVoiceBtn = document.getElementById('maleVoice');
        
        // Function to speak text
        function speakText(text, voiceType) {
            // Check if browser supports speech synthesis
            if ('speechSynthesis' in window) {
                // Stop any ongoing speech
                speechSynthesis.cancel();
                
                // Create a new speech synthesis utterance
                const utterance = new SpeechSynthesisUtterance();
                
                // Set the text to be spoken
                utterance.text = text;
                
                // Set voice properties based on selection
                if (voiceType === 'female') {
                    utterance.rate = 1.0;
                    utterance.pitch = 1.2;
                    utterance.volume = 1;
                } else {
                    utterance.rate = 1.0;
                    utterance.pitch = 0.8;
                    utterance.volume = 1;
                }
                
                // Get available voices and try to set a appropriate one
                const voices = speechSynthesis.getVoices();
                
                // Try to find a suitable voice
                let selectedVoice = null;
                
                if (voiceType === 'female') {
                    selectedVoice = voices.find(voice => 
                        voice.name.includes('Female') || 
                        voice.name.includes('woman') || 
                        voice.name.includes('Samantha') ||
                        voice.lang.includes('en-US')
                    );
                } else {
                    selectedVoice = voices.find(voice => 
                        voice.name.includes('Male') || 
                        voice.name.includes('man') || 
                        voice.name.includes('Alex') ||
                        voice.lang.includes('en-US')
                    );
                }
                
                if (selectedVoice) {
                    utterance.voice = selectedVoice;
                }
                
                // Speak the text
                speechSynthesis.speak(utterance);
                
                // Add visual feedback
                if (voiceType === 'female') {
                    femaleVoiceBtn.classList.add('pulse');
                    maleVoiceBtn.classList.remove('pulse');
                } else {
                    maleVoiceBtn.classList.add('pulse');
                    femaleVoiceBtn.classList.remove('pulse');
                }
                
                // Remove pulse animation when speech ends
                utterance.onend = function() {
                    femaleVoiceBtn.classList.remove('pulse');
                    maleVoiceBtn.classList.remove('pulse');
                };
            } else {
                alert('Your browser does not support text-to-speech functionality.');
            }
        }
        
        // Function to get all text content from the page
        function getAllPageText() {
            const contentElements = document.querySelectorAll('.content, .caption p, h1, h2, h3');

                allText += element.textContent + '. ';
            });
            
            return allText;
